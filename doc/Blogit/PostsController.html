def create
  @post = current_blogger.blog_posts.new(params[:post])
  if @post.save
    redirect_to @post, notice: t(:blog_post_was_successfully_created, scope: 'blogit.admin.posts')
  else
    render action: "new"
  end
end

def index
  respond_to do |format|
    format.xml {
      @posts = Post.order('created_at DESC')
    }
    format.html {
      @posts = Post.for_index(params[:page])
    }
    format.rss {
      @posts = Post.order('created_at DESC')
    }
  end
end

def update
  @post = current_blogger.blog_posts.find(params[:id])
  if @post.update_attributes(params[:post])
    redirect_to @post, notice: t(:blog_post_was_successfully_updated, scope: 'blogit.admin.posts')
  else
    render action: "edit"
  end
end
